; THIS EXA RECEIVES CC
; AND VERIFIES THE NUMS

MARK BEGRECVER
MAKE

MARK RECVERIFY
COPY M X
COPY M T
MARK RECPROCESS
;MODI X 2 T
TJMP VERODD
MARK ENDODDPRO
TEST X > 9
TJMP VERGREAT
MARK ENDGREAT
COPY X F
SEEK -9999
SEEK 15
TEST EOF
TJMP RECVERIFY
JUMP FINVERIFY


MARK VERODD
MULI X 2 X
JUMP ENDODDPRO

MARK VERGREAT
SUBI X 9 X
JUMP ENDGREAT


MARK FINVERIFY
SEEK -9999
COPY 0 X
MARK FINV
ADDI X F X
TEST EOF
FJMP FINV
SWIZ X 1 T ; 0 = VALID
COPY T M
FJMP VALIDNUMBER ; GOOD

WIPE ; RESTART
JUMP BEGRECVER


MARK VALIDNUMBER ; END
SEEK -9999
COPY 0 X
MARK CPVALNUM
COPY M F
ADDI X 1 X
TEST X = 16
FJMP CPVALNUM
