; THIS IS THE FILE COPY
; EXA THAT DISPATCHES

GRAB 300
COPY 8 X

MARK LOOPER
SEEK -9999
REPL MINION
MARK MINCOPY
COPY F M
TEST EOF
FJMP MINCOPY
COPY -9999 M
SUBI X 1 X
TEST X = 0
FJMP LOOPER
HALT


MARK MINION
MAKE
MARK MINCREATE
COPY M X
TEST X < 0
TJMP MINDIS
COPY X F
JUMP MINCREATE

MARK MINDIS
SEEK -9999
LINK 800
VOID M; DISPATCH
LINK 800
MARK MINFW
COPY F #DATA
TEST EOF
FJMP MINFW
; DONE WRITING
WIPE
MODE; LOCAL -> GLOBAL
COPY M X; WAIT TIME

MARK WAITLOOP
SUBI X 3 X
TEST X = 0
FJMP WAITLOOP
COPY 420 #PAGE ; EXECUTE
