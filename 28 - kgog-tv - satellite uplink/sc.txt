GRAB 300
LINK 800
LINK 799
SEEK 9999
SEEK -1
COPY F #FREQ

SEEK -9999
SEEK 1
COPY F X ; COPY AZIMUTH
REPL CHANGE_AZIM

SEEK 1
COPY F X ; COPY ELEVATIO
WIPE
JUMP CHANGE_ELEV

MARK CHANGE_AZIM
LINK 800
SUBI X #AZIM X
JUMP CHANGE_MOTOR

MARK CHANGE_ELEV
LINK 801
SUBI X #ELEV X

MARK CHANGE_MOTOR
TEST X < 0
TJMP MOTOR_DOWN
TEST X > 0
TJMP MOTOR_UP
JUMP END

MARK MOTOR_DOWN
COPY -1 #MOTR
ADDI X 1 X
TEST X = 0
FJMP MOTOR_DOWN
JUMP END


MARK MOTOR_UP
COPY 1 #MOTR
SUBI X 1 X
TEST X = 0
FJMP MOTOR_UP
JUMP END

MARK END
COPY 1 M
